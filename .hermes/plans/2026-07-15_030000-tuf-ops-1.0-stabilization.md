### Task 3: Server-Authoritative Identity

**Objective:** Frontend stores ONLY the JWT token. On startup, calls `/api/auth/me` for the authoritative user object.

**Files:**
- Modify: `apps/web/src/auth.ts` (remove localStorage user object storage)
- Modify: `apps/web/src/services/apiClient.ts` (read token, not user object)
- Modify: `apps/web/src/App.tsx` (call `/api/auth/me` on mount)

**Step 1: Change localStorage key semantics**
```typescript
// apps/web/src/auth.ts
const TOKEN_KEY = 'tuf_ops_token_v1'; // stores ONLY the JWT

export function persistToken(token: string) {
  localStorage.setItem(TOKEN_KEY, token);
}

export function getStoredToken(): string | null {
  return localStorage.getItem(TOKEN_KEY);
}

export function clearToken() {
  localStorage.removeItem(TOKEN_KEY);
}
```

**Step 2: Add fetchCurrentUser**
```typescript
// apps/web/src/auth.ts
export async function fetchCurrentUser(): Promise<AppUser | null> {
  const token = getStoredToken();
  if (!token) return null;
  try {
    const res = await fetch('/api/auth/me', {
      headers: { Authorization: *** ${token}`` }
    });
    if (!res.ok) return null;
    const data = await res.json();
    return normalizeApiUser(data.user);
  } catch {
    return null;
  }
}
```

**Step 3: Update apiClient to read token directly**
```typescript
// apps/web/src/services/apiClient.ts
function getAuthHeaders(): Record<string, string> {
  const token = getStoredToken();
  return token ? { Authorization: *** ${token}` } : {};
}
```

**Step 4: Update App.tsx startup**
```typescript
// apps/web/src/App.tsx
useEffect(() => {
  fetchCurrentUser().then(user => {
    if (user) setUser(user);
    else setUser(null);
  });
}, []);
```

**Step 5: Remove all localStorage user object reads**
Search for `getStoredUser()` across the codebase. Replace with `fetchCurrentUser()` or pass user via props/context.

**Step 6: Commit**
```bash
git add apps/web/src/auth.ts apps/web/src/services/apiClient.ts apps/web/src/App.tsx
git commit -m "refactor: server-authoritative identity — token-only localStorage"
```

**Verification:** Login → refresh page → user still authenticated. Role change in DB takes effect on next refresh.

**Step 1: Add legacy route logging**

```typescript
// apps/api/src/index.ts
server.addHook('onRequest', async (request) => {
  if (request.url.startsWith('/api/') && !request.url.startsWith('/api/v1/') && request.url !== '/api/auth/login') {
```

### Task 5: Mandatory CI Pipeline

- Create: `.github/workflows/ci.yml`

**Step 1: Create CI workflow**
```yaml
name: CI
on:
  pull_request:
    branches: [main]
  push:
    branches: [main]

jobs:
  verify:
    runs-on: ubuntu-latest
    services:
      postgres:
        image: postgres:16
        env:
          POSTGRES_USER: postgres
          POSTGRES_PASSWORD: postgres
          POSTGRES_DB: tuf_test
        ports: ['5432:5432']
    steps:
      - uses: actions/checkout@v4

      - run: pnpm install --frozen-lockfile
      - run: pnpm --filter @tuf/shared build
      - run: pnpm --filter api typecheck
      - run: pnpm --filter web typecheck
      - run: pnpm --filter web lint
      - run: pnpm --filter api lint
      - run: pnpm --filter api test
        env:
          DATABASE_URL: postgresql://postgres:postgres@localhost:5432/tuf_test
      - run: pnpm --filter web test
      - run: pnpm --filter web build
```

## Risks and Tradeoffs

| Risk | Mitigation |
|------|-----------|
| Removing `apps/frontend` breaks something in production | Archive to branch first, keep for 7 days before permanent deletion |
| Railway-only deployment has downtime during deploy | Railway zero-downtime deploys are sufficient for 6 users |
| Server-authoritative identity adds 1 API call on startup | `/api/auth/me` is a fast query — <50ms |
| Unified work_items migration is complex | Run in staging first, keep old tables as read-only during transition |
| CI adds friction to emergency fixes | Emergency bypass: admin can merge with failing CI — but it must be logged |
